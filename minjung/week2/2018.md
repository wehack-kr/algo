# #2018 - 수들의 합 5

- 택향님 풀이

```js
function solution(num) {
  let start = 0;
  let end = 0;
  let totalCount = 0;

  while (end <= num) {
    const partialSum =
      (end * (end + 1)) / 2 - (start * (start + 1)) / 2 + start;

    if (partialSum > num) {
      start += 1;
    } else {
      if (partialSum === num) {
        totalCount += 1;
      }
      end += 1;
    }
  }
  return totalCount;
}

```
