// 풀이 1.
function solution(nums) {
  for (let i = 0; i < nums.length - 1; i++) {
    for (let j = i + 1; j < nums.length; j++) {
      if (nums[i] > nums[j]) {
        let temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
      }
    }
  }

  let total = 0;
  let sum = 0;

  for (let k = 0; k < nums.length; k++) {
    total += sum + nums[k];
    sum += nums[k];
  }

  return total;
}

// 풀이 2.
function solution(nums) {
  return nums
    .sort((a, b) => a - b)
    .reduce((prev, curr, index) => {
      return prev + curr * (nums.length - index);
    }, 0);
}

const a = [3, 1, 4, 3, 2];
solution(a);
