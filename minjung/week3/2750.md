## 2750 수 정렬하기

```js
// 풀이 1
function solution(nums) {
  for (let i = 0; i < nums.length - 1; i++) {
    for (let j = i + 1; j < nums.length; j++) {
      if (nums[i] > nums[j]) {
        let temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
      }
    }
  }

  return nums;
}

// 풀이 2
function solution(nums) {
  return nums.sort((a, b) => a - b);
}

const a = [5, 2, 3, 4, 1];
solution(a);
```
