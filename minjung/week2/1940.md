# #1940 - 주몽

```js
function solution(args) {
  const [_, m, uniqIds] = args;

  const M = Number(m);
  const sortedUniqIds = uniqIds
    .split(' ')
    .map(Number)
    .sort((a, b) => a - b);

  let left = 0;
  let right = sortedUniqIds.length - 1;
  let count = 0;

  while (left < right) {
    let sum = sortedUniqIds[left] + sortedUniqIds[right];

    if (sum === M) {
      count++;
      left++;
      right--;
    } else if (sum < M) {
      left++;
    } else {
      right--;
    }
  }

  return count;
}
// 예
// const a = ['6', '9', '2 7 4 1 5 3'];
// solution(a);
```
