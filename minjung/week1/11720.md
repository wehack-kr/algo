# #11720 - 숫자의 합

```js
function solution(args) {
  const value = args[1];

  let result = 0;

  for (let i = 0; i < value.length; i++) {
    result += Number(value[i]);
  }

  return result;
}

// 예
// const a = [1, '1'];
// solution(a);
```

## 제출용

```js
var fs = require('fs');
var input = fs.readFileSync('/dev/stdin').toString().split('\n');

const value = input[1];

  result += Number(value[i]);
```
